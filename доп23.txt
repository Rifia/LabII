Вам нужны функции открытия файла, чтения и закрытия. 
Открываете файл, читаете построчно и увеличивайте счетчик. 
Когда счетчик достиг нужной, выводим строку. схематический код:

char * getLine(char* filename, int nedded) {
  char line[250];
  FILE * f = open(filename, "r");
  int c = 0;
  while (c < needed && not eof(f)) {
     fscanf(f, "%s", line);
     c++;
  }
  close(f);
  if (c!= nedded) {
    return NULL;
  }
  return strdup(line); // не забыть потом освободить буфер
}
